        // CONFIG

        // NAVIGATION
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
            const target = document.getElementById('tab-' + tabId);
            if (target) target.classList.add('active');
            
            document.querySelectorAll('.nav-btn').forEach(btn => {
                if(btn.dataset.target === tabId) {
                    btn.classList.add('text-blue-600', 'scale-110'); btn.classList.remove('text-slate-400');
                } else {
                    btn.classList.remove('text-blue-600', 'scale-110'); btn.classList.add('text-slate-400');
                }
            });
            document.getElementById('main-scroll').scrollTop = 0;
        }

        document.querySelectorAll('.nav-btn').forEach(btn => btn.addEventListener('click', (e) => switchTab(e.currentTarget.dataset.target)));

        // LOGIQUE PERFORMANCE
        let chartInstance = null;
        let rideHistory = []; 
        let globalDailyData = [];
        let currentChartRange = '30';

        function parseNum(v) { if(!v) return 0; let s = v.toString().replace(',','.'); let n = parseFloat(s); return isNaN(n)?0:n; }

        function processData(activities) {
            let workoutMap = {}; let minDate = new Date('2099-01-01'); let maxDate = new Date('2000-01-01');
            rideHistory = []; 
            const getVal = (o, kws) => { for(let k in o) { for(let kw of kws) { if(k.includes(kw)) return o[k]; } } return null; };
            
            let parsed = [];
            activities.forEach(act => {
                let n = {}; for(let k in act) { if(k) n[k.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g,"").replace(/[^a-z0-9]/g,"")] = act[k]; }
                let type = (getVal(n, ['type']) || "").toLowerCase();
                if(type && !type.includes('ride') && !type.includes('velo') && !type.includes('cycl')) return;
                let dStr = getVal(n, ['date']); if(!dStr) return;
                let d = new Date(dStr);
                if(isNaN(d.getTime())) {
                    let m = dStr.match(/\b(\d{1,4})[/\.\-](\d{1,2})[/\.\-](\d{1,4})\b/);
                    if(m) { let p1=parseInt(m[1]), p2=parseInt(m[2]), p3=parseInt(m[3]); if(p1>1000) d=new Date(p1, p2-1, p3); else d=new Date(p3, p2-1, p1); }
                }
                if(isNaN(d.getTime())) return;
                n._d = d; parsed.push(n);
            });

            parsed.sort((a,b)=>a._d - b._d);
            parsed.forEach(n => {
                let dist = parseNum(getVal(n,['distance']));
                let ele = parseNum(getVal(n,['elevation','denivele']));
                let hr = parseNum(getVal(n,['heartrate','fcmoyenne']));
                let time = parseNum(getVal(n,['movingtime','temps']));
                if(dist>1000 && time>0 && (dist/time)<40) dist/=1000;
                let tss = parseFloat(getVal(n,['effort','tss','suffer']));
                if(isNaN(tss) || tss <= 0) tss = (dist*1.3) + (ele*0.04);
                let dk = n._d.toISOString().split('T')[0];
                workoutMap[dk] = (workoutMap[dk]||0)+tss;
                if(n._d < minDate) minDate = n._d; if(n._d > maxDate) maxDate = n._d;
                if(dist>5 && time>0) rideHistory.push({date:dk, dist, ele, hr, speed:dist/(time/3600), ratio:ele/dist});
            });

            let daily = []; let ctl = 0, atl = 0;
            let now = new Date(); now.setHours(0,0,0,0);
            if(maxDate < now) maxDate = now;
            let limit = new Date(maxDate); limit.setDate(limit.getDate()+14);
            let cur = new Date(minDate);
            while(cur <= limit) {
                let dk = cur.toISOString().split('T')[0];
                let tss = workoutMap[dk]||0;
                ctl += (tss - ctl)/42; atl += (tss - atl)/7;
                daily.push({date:dk, ctl, atl, tsb:ctl-atl, isFuture:cur > now});
                rideHistory.filter(r=>r.date===dk).forEach(r=>{r.ctl=ctl; r.tsb=ctl-atl;});
                cur.setDate(cur.getDate()+1);
            }
            return daily;
        }

        function updateUI(daily) {
            if(!daily.length) return;
            globalDailyData = daily;
            const nowData = daily.filter(d=>!d.isFuture).pop();
            const tsb = Math.round(nowData.tsb);
            document.getElementById('valCTL').innerText = Math.round(nowData.ctl);
            document.getElementById('valATL').innerText = Math.round(nowData.atl);
            document.getElementById('valTSB').innerText = (tsb>0?'+':'')+tsb;
            const icon = document.getElementById('statusIcon');
            const label = document.getElementById('statusLabel');
            if(tsb < -25) { icon.innerText="🛋️"; label.innerText="Repos requis"; label.className="text-2xl font-black text-red-600"; }
            else if(tsb < -10) { icon.innerText="📈"; label.innerText="Productif"; label.className="text-2xl font-black text-blue-600"; }
            else if(tsb < 10) { icon.innerText="✅"; label.innerText="Optimal"; label.className="text-2xl font-black text-green-600"; }
            else { icon.innerText="🚀"; label.innerText="Affûté (Peak)"; label.className="text-2xl font-black text-yellow-500"; }
            updateChartDisplay();
        }

        function updateChartDisplay() {
            const actual = globalDailyData.filter(d=>!d.isFuture);
            let disp = globalDailyData;
            if(currentChartRange !== 'all') disp = globalDailyData.slice(Math.max(0, actual.length - parseInt(currentChartRange)));
            const relToday = disp.filter(d=>!d.isFuture).length - 1;
            const fmt = (s) => { let p=s.split('-'); return `${p[2]}/${p[1]}`; };
            const ctx = document.getElementById('fitnessChart').getContext('2d');
            if(chartInstance) chartInstance.destroy();
            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: disp.map(d=>d.isFuture?fmt(d.date)+'+':fmt(d.date)),
                    datasets: [
                        { label:'CTL', data:disp.map(d=>d.ctl), borderColor:'#3b82f6', backgroundColor:'rgba(59,130,246,0.1)', fill:true, pointRadius:0, tension:0.4, segment:{borderDash:c=>c.p0DataIndex>=relToday?[5,5]:undefined} },
                        { label:'ATL', data:disp.map(d=>d.atl), borderColor:'#ef4444', pointRadius:0, tension:0.4, segment:{borderDash:c=>c.p0DataIndex>=relToday?[5,5]:undefined} },
                        { label:'TSB', data:disp.map(d=>d.tsb), borderColor:'#eab308', pointRadius:0, tension:0.4, segment:{borderDash:c=>c.p0DataIndex>=relToday?[5,5]:undefined} }
                    ]
                },
                options: { 
                    responsive:true, maintainAspectRatio:false, plugins:{legend:{display:false}},
                    scales:{ x:{ticks:{maxTicksLimit:5, font:{size:9}}}, y:{ticks:{font:{size:9}}} }
                }
            });
        }

        function loadDemo() {
            const data = []; let d = new Date(); d.setDate(d.getDate()-150);
            for(let i=0; i<150; i++) {
                let dow=d.getDay(), tss=0; if(dow===0) tss=125; else if(dow===2) tss=80; else if(dow===4) tss=60; else if(dow===6) tss=95;
                if(tss>0 && Math.random()>0.15) data.push({date:d.toISOString().split('T')[0], tss, distance:tss*0.65, elevation:tss*10, type:'Ride'});
                d.setDate(d.getDate()+1);
            }
            updateUI(processData(data));
        }

        document.getElementById('btnLoadSample').addEventListener('click', loadDemo);
        document.getElementById('csvFileInput').addEventListener('change', (e) => {
            Papa.parse(e.target.files[0], { header:true, complete: r => updateUI(processData(r.data)) });
        });

        document.getElementById('btnPredict').addEventListener('click', () => {
            if(!rideHistory.length) return alert("Données requises.");
            const d=parseFloat(document.getElementById('predDist').value), e=parseFloat(document.getElementById('predDeniv').value), r=e/d;
            const now=globalDailyData.filter(d=>!d.isFuture).pop();
            let scored = rideHistory.map(h => ({...h, s:(Math.abs(h.ratio-r)*6) + (Math.abs(h.dist-d)/d*15) })).sort((a,b)=>a.s-b.s).slice(0,5);
            let spd = 0; scored.forEach(h => { spd += h.speed * Math.max(0.85, Math.min(1.15, 1+((now.ctl-h.ctl)/300)+((now.tsb-h.tsb)/450))); }); spd /= scored.length;
            const t = d/spd; const res = document.getElementById('predResult');
            res.innerHTML = `<div class="bg-blue-600 text-white p-7 rounded-[2rem] text-center shadow-xl"><div class="text-5xl font-black">${Math.floor(t)}h ${Math.round((t-Math.floor(t))*60)}m</div></div>`;
            res.classList.remove('hidden');
        });

        document.getElementById('btnCoach').addEventListener('click', () => {
            if(!globalDailyData.length) return alert("Données requises.");
            const goal = document.getElementById('goalSelector').value;
            const nowIdx = globalDailyData.filter(d=>!d.isFuture).length - 1;
            let targetTsb = (goal === 'pma') ? -5 : (goal === 'affutage' ? 10 : -15);
            let day = globalDailyData.slice(nowIdx).find(d=>d.tsb >= targetTsb) || globalDailyData.pop();
            let diff = globalDailyData.indexOf(day) - nowIdx;
            const res = document.getElementById('coachResult');
            res.innerHTML = `<div class="bg-slate-900 text-white p-6 rounded-[2rem] shadow-xl"><h4 class="text-2xl font-black mb-1">${diff===0?'Aujourd\'hui':(diff===1?'Demain':'Le '+day.date.split('-')[2]+'/'+day.date.split('-')[1])}</h4><p class="text-xs opacity-60">TSB estimé : ${Math.round(day.tsb)}</p></div>`;
            res.classList.remove('hidden');
        });

        document.querySelectorAll('.range-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                document.querySelectorAll('.range-btn').forEach(b => { b.classList.remove('bg-white', 'shadow-sm', 'text-slate-800'); b.classList.add('text-slate-500'); });
                e.target.classList.add('bg-white', 'shadow-sm', 'text-slate-800'); e.target.classList.remove('text-slate-500');
                currentChartRange = e.target.dataset.range;
                updateChartDisplay();
            });
        });

        // INIT
        window.onload = function() {
            loadDemo();
            switchTab('dashboard');
        };
