        function parseNum(v) { if(!v) return 0; let s = v.toString().replace(',','.'); let n = parseFloat(s); return isNaN(n)?0:n; }

        function processData(activities) {
            let workoutMap = {}; let minDate = new Date('2099-01-01'); let maxDate = new Date('2000-01-01');
            rideHistory = []; 
            const getVal = (o, kws) => { for(let k in o) { for(let kw of kws) { if(k.includes(kw)) return o[k]; } } return null; };

            let parsed = [];
            activities.forEach(act => {
                let n = {}; for(let k in act) { if(k) n[k.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g,"").replace(/[^a-z0-9]/g,"")] = act[k]; }
                let type = (getVal(n, ['type']) || "").toLowerCase();
                if(type && !type.includes('ride') && !type.includes('velo') && !type.includes('cycl')) return;
                
                let dStr = getVal(n, ['date']); if(!dStr) return;
                let d = new Date(dStr);
                if(isNaN(d.getTime())) {
                    let m = dStr.match(/\b(\d{1,4})[/\.\-](\d{1,2})[/\.\-](\d{1,4})\b/);
                    if(m) {
                        let p1=parseInt(m[1]), p2=parseInt(m[2]), p3=parseInt(m[3]);
                        if(p1>1000) d=new Date(p1, p2-1, p3); else d=new Date(p3, p2-1, p1);
                    }
                }
                if(isNaN(d.getTime())) return;
                n._d = d; parsed.push(n);
            });

            parsed.sort((a,b)=>a._d - b._d);
            parsed.forEach(n => {
                let dist = parseNum(getVal(n,['distance']));
                let ele = parseNum(getVal(n,['elevation','denivele']));
                let hr = parseNum(getVal(n,['heartrate','fcmoyenne']));
                let time = parseNum(getVal(n,['movingtime','temps']));
                if(dist>1000 && time>0 && (dist/time)<30) dist/=1000;
                let tss = parseFloat(getVal(n,['effort','tss','suffer']));
                if(isNaN(tss) || tss <= 0) tss = (dist*1.4) + (ele*0.04);
                
                let dk = n._d.toISOString().split('T')[0];
                workoutMap[dk] = (workoutMap[dk]||0)+tss;
                if(n._d < minDate) minDate = n._d; if(n._d > maxDate) maxDate = n._d;
                if(dist>5 && time>0) rideHistory.push({date:dk, dist, ele, hr, speed:dist/(time/3600), ratio:ele/dist});
            });

            let daily = []; let ctl = 0, atl = 0;
            let now = new Date(); now.setHours(0,0,0,0);
            if(maxDate < now) maxDate = now;
            let limit = new Date(maxDate); limit.setDate(limit.getDate()+14);
            let cur = new Date(minDate);
            while(cur <= limit) {
                let dk = cur.toISOString().split('T')[0];
                let tss = workoutMap[dk]||0;
                ctl += (tss - ctl)/42; atl += (tss - atl)/7;
                daily.push({date:dk, ctl, atl, tsb:ctl-atl, isFuture:cur > now});
                rideHistory.filter(r=>r.date===dk).forEach(r=>{r.ctl=ctl; r.tsb=ctl-atl;});
                cur.setDate(cur.getDate()+1);
            }
            return daily;
        }

        function updateChartDisplay() {
            const actual = globalDailyData.filter(d=>!d.isFuture);
            let disp = globalDailyData;
            if(currentChartRange !== 'all') disp = globalDailyData.slice(Math.max(0, actual.length - parseInt(currentChartRange)));
            const relToday = disp.filter(d=>!d.isFuture).length - 1;
            const fmt = (s) => { let p=s.split('-'); return `${p[2]}/${p[1]}`; };
            
            const ctx = document.getElementById('fitnessChart').getContext('2d');
            if(chartInstance) chartInstance.destroy();
            chartInstance = new Chart(ctx, {
                type: 'line',
                data: {

        document.getElementById('btnPredict').addEventListener('click', () => {
            if(!rideHistory.length) return notify("Données requises.");
            const d=parseFloat(document.getElementById('predDist').value);
            const e=parseFloat(document.getElementById('predDeniv').value);
            const r=e/d; const now=globalDailyData.filter(d=>!d.isFuture).pop();
            let scored = rideHistory.map(h => ({...h, s:(Math.abs(h.ratio-r)*5) + (Math.abs(h.dist-d)/d*10) })).sort((a,b)=>a.s-b.s).slice(0,5);
            let spd = 0;
            scored.forEach(h => {
                let mult = 1 + ((now.ctl-h.ctl)/300) + ((now.tsb-h.tsb)/400);
                spd += h.speed * Math.max(0.85, Math.min(1.15, mult));
            });
            spd /= scored.length;
            const t = d/spd; const hrs=Math.floor(t), mins=Math.round((t-hrs)*60);
            const res = document.getElementById('predResult');
            res.innerHTML = `<div class="bg-blue-600 text-white p-6 rounded-3xl shadow-xl text-center"><p class="text-[10px] font-black uppercase opacity-70 mb-1">Chrono Estimé IA</p><div class="text-4xl font-black mb-1">${hrs}h ${mins.toString().padStart(2,'0')}m</div><p class="text-xs opacity-80">Vitesse moy. ajustée : ${spd.toFixed(1)} km/h</p></div>`;
            res.classList.remove('hidden');
        });

        document.getElementById('btnLoadSample').addEventListener('click', () => {
            const data = []; let d = new Date(); d.setDate(d.getDate()-180);
            for(let i=0; i<180; i++) {
                let dow=d.getDay(), tss=0;
                if(dow===0) tss=120; else if(dow===3) tss=80; else if(dow===6) tss=60;
                if(tss>0 && Math.random()>0.2) data.push({date:d.toISOString().split('T')[0], tss, distance:tss*0.7, elevation:tss*10, type:'Ride'});
                d.setDate(d.getDate()+1);
            }
            updateUI(processData(data));
            notify("Mode Démo activé", "success");
        });
